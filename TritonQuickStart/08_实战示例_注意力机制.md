## 8.1 Flash Attention 原理

标准 Attention 计算：`O = softmax(Q @ K^T / sqrt(d)) @ V`

Flash Attention 的核心优化：**分块计算**，避免实例化完整的 N x N 注意力矩阵。

```
标准 Attention:
  Q [M, d] @ K^T [d, N] -> S [M, N]  (完整注意力矩阵)
  softmax(S) -> P [M, N]
  P @ V [N, d] -> O [M, d]

Flash Attention (分块):
  对 K, V 按 BLOCK_N 分块:
    Q [BLOCK_M, d] @ K_block^T [d, BLOCK_N] -> S_block [BLOCK_M, BLOCK_N]
    在线 softmax 更新: m_i, l_i, O_i
    避免存储完整 [M, N] 矩阵
```

### 在线 Softmax

关键技巧：维护运行中的最大值 `m_i` 和指数和 `l_i`，逐块更新输出 `O_i`。

```
初始: m_i = -inf, l_i = 0, O_i = 0

处理第 j 个 K,V 分块:
  1. S_ij = Q_i @ K_j^T / sqrt(d)         # 当前块注意力分数
  2. m_ij = max(S_ij, axis=-1)             # 当前块最大值
  3. m_new = max(m_i, m_ij)                # 更新全局最大值
  4. P_ij = exp(S_ij - m_new)              # 当前块 softmax 分子
  5. l_new = exp(m_i - m_new) * l_i + sum(P_ij)  # 更新指数和
  6. O_i = exp(m_i - m_new) * O_i * l_i / l_new + P_ij @ V_j / l_new  # 更新输出
  7. m_i = m_new, l_i = l_new
```

## 8.2 Triton 实现

```python
import torch
import torch_npu
import triton
import triton.language as tl
import math

@triton.jit
def flash_attn_fwd_kernel(
    Q_ptr, K_ptr, V_ptr, O_ptr,
    # 维度
    Z, H, M, N_CTX, HEAD_DIM: tl.constexpr,
    # 步长 (Q: [Z, H, M, D], K/V: [Z, H, N, D], O: [Z, H, M, D])
    stride_qz, stride_qh, stride_qm, stride_qd,
    stride_kz, stride_kh, stride_kn, stride_kd,
    stride_vz, stride_vh, stride_vn, stride_vd,
    stride_oz, stride_oh, stride_om, stride_od,
    # 其他参数
    scale,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr,
):
    # 获取当前 program 的任务索引
    pid = tl.program_id(0)

    # 将一维 pid 还原为 (batch, head, m_block) 三维索引
    num_m_blocks = (M + BLOCK_M - 1) // BLOCK_M
    batch_head_idx = pid // num_m_blocks
    m_block_idx = pid % batch_head_idx  # 简化，实际为 pid % num_m_blocks

    off_z = batch_head_idx // H
    off_h = batch_head_idx % H
    off_m = m_block_idx * BLOCK_M + tl.arange(0, BLOCK_M)

    # Q 的基址偏移
    q_base = off_z.to(tl.int64) * stride_qz + off_h.to(tl.int64) * stride_qh

    # 加载 Q 分块 [BLOCK_M, HEAD_DIM]
    q_offsets = off_m[:, None] * stride_qm + tl.arange(0, HEAD_DIM)[None, :] * stride_qd
    q_mask = (off_m[:, None] < M) & (tl.arange(0, HEAD_DIM)[None, :] < HEAD_DIM)
    Q = tl.load(Q_ptr + q_base + q_offsets, mask=q_mask, other=0.0)

    # 初始化在线 softmax 状态
    m_i = tl.full([BLOCK_M], float('-inf'), dtype=tl.float32)  # 行最大值
    l_i = tl.zeros([BLOCK_M], dtype=tl.float32)                 # 行指数和
    O_acc = tl.zeros([BLOCK_M, HEAD_DIM], dtype=tl.float32)     # 输出累加器

    # K, V 的基址偏移
    kv_base = off_z.to(tl.int64) * stride_kz + off_h.to(tl.int64) * stride_kh

    # 逐块遍历 K, V
    for start_n in range(0, N_CTX, BLOCK_N):
        off_n = start_n + tl.arange(0, BLOCK_N)

        # 加载 K 分块 [BLOCK_N, HEAD_DIM]
        k_offsets = off_n[:, None] * stride_kn + tl.arange(0, HEAD_DIM)[None, :] * stride_kd
        k_mask = (off_n[:, None] < N_CTX) & (tl.arange(0, HEAD_DIM)[None, :] < HEAD_DIM)
        K_block = tl.load(K_ptr + kv_base + k_offsets, mask=k_mask, other=0.0)

        # 计算 S = Q @ K^T / sqrt(d)
        # Q: [BLOCK_M, HEAD_DIM], K^T: [HEAD_DIM, BLOCK_N]
        S = tl.dot(Q, K_block.trans()) * scale  # [BLOCK_M, BLOCK_N]

        # 更新在线 softmax
        m_ij = tl.max(S, axis=1)             # [BLOCK_M]
        m_new = tl.maximum(m_i, m_ij)        # [BLOCK_M]
        alpha = tl.exp(m_i - m_new)           # [BLOCK_M]
        beta = tl.exp(m_ij - m_new)           # [BLOCK_M]
        l_new = alpha * l_i + beta * tl.sum(tl.exp(S - m_ij[:, None]), axis=1)

        # 加载 V 分块 [BLOCK_N, HEAD_DIM]
        v_offsets = off_n[:, None] * stride_vn + tl.arange(0, HEAD_DIM)[None, :] * stride_vd
        v_mask = (off_n[:, None] < N_CTX) & (tl.arange(0, HEAD_DIM)[None, :] < HEAD_DIM)
        V_block = tl.load(V_ptr + kv_base + v_offsets, mask=v_mask, other=0.0)

        # 更新输出累加器
        # P = exp(S - m_ij[:, None])  [BLOCK_M, BLOCK_N]
        P = tl.exp(S - m_ij[:, None])
        # O_acc = alpha[:, None] * O_acc * l_i[:, None] / l_new[:, None] + (P @ V_block) / l_new[:, None]
        O_acc = alpha[:, None] * O_acc * l_i[:, None] + tl.dot(P, V_block)
        O_acc = O_acc / l_new[:, None]

        # 更新状态
        m_i = m_new
        l_i = l_new

    # 最终输出
    o_offsets = off_m[:, None] * stride_om + tl.arange(0, HEAD_DIM)[None, :] * stride_od
    o_base = off_z.to(tl.int64) * stride_oz + off_h.to(tl.int64) * stride_oh
    o_mask = (off_m[:, None] < M) & (tl.arange(0, HEAD_DIM)[None, :] < HEAD_DIM)
    tl.store(O_ptr + o_base + o_offsets, O_acc, mask=o_mask)


def flash_attention(Q, K, V, BLOCK_M=64, BLOCK_N=64):
    Z, H, M, D = Q.shape
    _, _, N, _ = K.shape
    scale = 1.0 / math.sqrt(D)
    O = torch.empty_like(Q)

    # 使用固定核数
    import triton.runtime.driver as driver
    device = torch.npu.current_device()
    props = driver.active.utils.get_device_properties(device)
    aicore_num = props["num_aicore"]

    num_m_blocks = triton.cdiv(M, BLOCK_M)
    total_tasks = Z * H * num_m_blocks
    grid = (min(total_tasks, aicore_num),)

    flash_attn_fwd_kernel[grid](
        Q, K, V, O,
        Z, H, M, N, D,
        Q.stride(0), Q.stride(1), Q.stride(2), Q.stride(3),
        K.stride(0), K.stride(1), K.stride(2), K.stride(3),
        V.stride(0), V.stride(1), V.stride(2), V.stride(3),
        O.stride(0), O.stride(1), O.stride(2), O.stride(3),
        scale,
        BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N,
    )
    return O
```

## 8.3 UB 溢出预防

Flash Attention 的片上内存占用较大，需要注意：

```
单个 program 的 UB 占用估算:

Q 分块:   BLOCK_M * HEAD_DIM * dtype_size
K 分块:   BLOCK_N * HEAD_DIM * dtype_size
V 分块:   BLOCK_N * HEAD_DIM * dtype_size
S 矩阵:  BLOCK_M * BLOCK_N * 4 (float32)
O 累加器: BLOCK_M * HEAD_DIM * 4 (float32)
m_i:      BLOCK_M * 4
l_i:      BLOCK_M * 4

总计 ≈ (BLOCK_M + 2*BLOCK_N) * HEAD_DIM * dtype_size
     + BLOCK_M * BLOCK_N * 4
     + BLOCK_M * HEAD_DIM * 4
```

**安全配置示例（UB ~192KB）：**

```python
# HEAD_DIM=64, float16 (2 bytes)
# BLOCK_M=64, BLOCK_N=64
# UB ≈ (64+128)*64*2 + 64*64*4 + 64*64*4
#     = 24576 + 16384 + 16384 = 57344 bytes ≈ 56KB ✓

# HEAD_DIM=128, float16
# BLOCK_M=32, BLOCK_N=32
# UB ≈ (32+64)*128*2 + 32*32*4 + 32*128*4
#     = 24576 + 4096 + 16384 = 45056 bytes ≈ 44KB ✓
```

## 8.4 测试验证

```python
# 生成测试数据
Z, H, M, N, D = 1, 4, 256, 256, 64
Q = torch.randn(Z, H, M, D, device="npu", dtype=torch.float16)
K = torch.randn(Z, H, N, D, device="npu", dtype=torch.float16)
V = torch.randn(Z, H, N, D, device="npu", dtype=torch.float16)

# Triton-Ascend Flash Attention
O_triton = flash_attention(Q, K, V, BLOCK_M=64, BLOCK_N=64)

# PyTorch 参考
scale = 1.0 / math.sqrt(D)
S = torch.matmul(Q, K.transpose(-2, -1)) * scale
P = torch.softmax(S, dim=-1)
O_ref = torch.matmul(P, V)

# 比较精度
diff = torch.max(torch.abs(O_triton.float() - O_ref.float())).item()
print(f"最大误差: {diff}")
print(f"验证: {'PASS' if diff < 0.1 else 'FAIL'}")  # float16 注意力误差较宽容
```
